  public void Key_()
  {
    if (this.t < 50)
      return;
    if (Input.GetButtonDown("Fire5"))
    {
      this.t = 0;
      if (this.selectchar > 0)
        --this.selectchar;
      this.OnClick(this.selectchar);

    else if (Input.GetButtonDown("Fire6"))

      this.t = 0;
      if (this.selectchar < 4)
        ++this.selectchar;
      this.OnClick(this.selectchar);
    }
    else if (Input.GetButtonDown("Fire3"))
